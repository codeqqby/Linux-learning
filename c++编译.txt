方法一：
1. 代码:
#include <iostream>
using namespace std;
int main()
{
    cout<<"Hello,World!\n"<<endl;
    return 0;
}


2. 存盘为Hello.cpp

3. g++ Hello.cpp -o hello

4. ./hello

方法二：
1. 代码:
//file1:Hello.h
class Hello {
public:
     Hello();
     void Display();
};
//file2:Hello.cpp
#include <iostream>
#include "Hello.h"
using namespace std;
Hello::Hello()
{
}
void Hello::Display()
{
    cout<<"Hello,World!\n"<<endl;
}
//file3:H.cpp
#include <iostream>
#include "Hello.h"
int main()
{
    Hello h;
    h.Display();
    return 0;
}

2. 按顺序编译文件(参数-c 只编译不连接)
g++ -c Hello.cpp -o Hello.o
g++ -c H.cpp -o H.o
g++ H.o Hello.o -o H

3. ./H

方法三：
1. 代码同上

2. Makefile
H : H.o Hello.o
	g++ H.o Hello.o -o H
Hello.o : Hello.cpp
	g++ -c Hello.cpp -o Hello.o
H.o : H.cpp
	g++ -c H.cpp -o H.o
clean : 
	rm H H.o Hello.o

3. make

4. ./H

5. make clean
